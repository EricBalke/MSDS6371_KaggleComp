/* Create dummy variables for the Neighborhoods */
data train_2 (Keep = Id Neighborhood SalePrice GrLIvArea N_Dummy E_Dummy int_e int_n);

run;

proc univariate data = train_2;
run;

run;
title;

proc print data= train_2 (obs = 10);
run;

ods graphics on;

	model SalePrice = GrLIvArea E_Dummy N_Dummy int_e int_n/CLM R  CLB VIF;
	ID id;
